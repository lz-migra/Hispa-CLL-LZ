¡Claro que sí! 😎🔥  
Podemos hacer que el script **inyecte emojis según el tipo de beneficio detectado** en la descripción o el nombre. Por ejemplo:

- Si encuentra **GB / Internet → 🌐**
    
- Si encuentra **MIN / minutos → 📞**
    
- Si encuentra **SMS → ✉️**
    
- Si encuentra **CUP / saldo → 💵**
    

Aquí te dejo el script mejorado con lógica de emojis:

```javascript
// == Cuballama Catalog Fetcher ==
// Este script obtiene el catálogo de ofertas de Cuballama, lo formatea
// y lo muestra en un panel flotante en la página. Es inmune a Darkreader.
// Para usar: Pega este código en la consola de desarrollador del navegador.
// -----------------------------------------------------------------------------

(async function() {
  'use strict';

  // -------------------------------------------------
  // 🔗 CONFIGURACIÓN
  // -------------------------------------------------
  const API_URL = "https://api.cuballama.com/catalog/offers/cll/TOPUP?countryCode=CUB&operator=CUBACEL";
  const TOKEN_KEY = "token_cuballama";
  const PANEL_ID = "cuballama-catalog-panel";

  // -------------------------------------------------
  // 🔐 UTILIDADES DE DATOS
  // -------------------------------------------------

  /**
   * Normaliza el token de autenticación, añadiendo "Bearer " si es necesario.
   * @param {string | null} token - El token sin procesar.
   * @returns {string | null} El token normalizado o null si no existe.
   */
  function normalizeAuth(token) {
    if (!token) return null;
    const trimmedToken = token.trim();
    return trimmedToken.toLowerCase().startsWith("bearer ") ? trimmedToken : `Bearer ${trimmedToken}`;
  }

  /**
   * Extrae el array de ofertas desde el objeto de respuesta de la API,
   * probando varias rutas comunes de forma segura.
   * @param {object} data - El objeto JSON de la respuesta de la API.
   * @returns {Array<object> | null} El array de ofertas o null si no se encuentra.
   */
  function getOffersFromData(data) {
    return data?.offers ?? data?.patch?.offers ?? data?.data?.offers ?? null;
  }

  // -------------------------------------------------
  // 🎨 PROCESAMIENTO Y FORMATO
  // -------------------------------------------------

  /**
   * Aplica emojis a fragmentos de texto relevantes en la descripción de la oferta.
   * @param {string} text - El texto a procesar.
   * @returns {string} El texto con emojis.
   */
  function addEmojis(text) {
    if (!text) return "";
    return String(text)
      .replace(/(\d+\s*GB)/gi, '🌐 $1')
      .replace(/(\d+\s*MIN|\bminutos?\b)/gi, '📞 $1')
      .replace(/(\d+\s*SMS)/gi, '✉️ $1')
      .replace(/(\d+\s*(?:CUP|USD|EUR))/gi, '💵 $1')
      .replace(/\b(Ilimitado|ilimitada|Ilimitados|ilimitados)\b/gi, '⚡ $1');
  }

  /**
   * Normaliza una única oferta a un formato estándar { nombre, descripcion, precio }.
   * @param {object} offer - El objeto de oferta de la API.
   * @returns {{nombre: string, descripcion: string, precio: string}}
   */
  function normalizeOffer(offer) {
    const nombre = offer.title ?? offer.name ?? offer.skus?.[0] ?? "Sin título";
    
    let descripcion = offer.subTitle ?? offer.description ?? "";
    if (!descripcion && Array.isArray(offer.productItems) && offer.productItems.length > 0) {
      descripcion = offer.productItems.map(pi => pi.name || pi.sku).filter(Boolean).join(" • ");
    }
    
    const priceValue = offer.finalPrice ?? offer.generalPrice ?? offer.price ?? offer.saleValue ?? null;
    const currency = (offer.currency ?? offer.currencyCode ?? "USD").toUpperCase();
    let precio = "—";

    // ✨ LÓGICA DE FORMATO DE PRECIO ACTUALIZADA ✨
    if (priceValue !== null) {
      switch (currency) {
        case "USD":
          precio = `$ ${priceValue}`;
          break;
        case "EUR":
          precio = `${priceValue} €`;
          break;
        case "CAD":
          precio = `CAD ${priceValue}`;
          break;
        default: // Formato para cualquier otra moneda (ej. CUP)
          precio = `${priceValue} ${currency}`;
          break;
      }
    }
    
    return {
      nombre: String(nombre),
      descripcion: addEmojis(String(descripcion)),
      precio: String(precio),
    };
  }

  /**
   * Formatea una lista de ofertas normalizadas a un texto plano para visualización.
   * @param {Array<{nombre: string, descripcion: string, precio: string}>} offers
   * @returns {string} El catálogo formateado como texto.
   */
  function formatOffersToText(offers) {
    // ✨ SALTO DE LÍNEA ADICIONAL AÑADIDO AQUÍ (\n\n) ✨
    const header = "📱 PLANES Y RECARGAS DISPONIBLES 📱\n\n";
    const body = offers.map((oferta, index) => 
      `${index + 1}. *${oferta.nombre}*\n${oferta.descripcion}\n*${oferta.precio}*`
    ).join("\n\n---\n\n");

    return header + body;
  }

  // -------------------------------------------------
  // 🪟 INTERFAZ DE USUARIO (PANEL)
  // -------------------------------------------------

  /**
   * Muestra un panel flotante con el catálogo y un botón para copiar.
   * @param {string} textContent - El texto a mostrar en el panel.
   */
  function displayCatalogPanel(textContent) {
    document.getElementById(PANEL_ID)?.remove();

    const panelCSS = `
      position: fixed; right: 20px; bottom: 20px; width: 360px; max-height: 70vh;
      background: #0d1117; color: #c9d1d9; border: 1px solid #30363d;
      border-radius: 8px; box-shadow: 0 10px 30px rgba(0,0,0,0.5);
      z-index: 999999; font-family: 'SF Mono', Consolas, 'Courier New', monospace;
      font-size: 13px; line-height: 1.4; display: flex; flex-direction: column;
    `;
    const headerCSS = `
      display: flex; justify-content: space-between; align-items: center;
      padding: 10px 15px; background: #161b22; border-bottom: 1px solid #30363d;
      font-weight: 600;
    `;
    const contentCSS = `padding: 15px; overflow-y: auto; white-space: pre-wrap;`;
    const buttonCSS = `
      background: #21262d; color: #c9d1d9; border: 1px solid #30363d;
      border-radius: 6px; padding: 5px 10px; cursor: pointer; margin-left: 8px;
      transition: background-color 0.2s;
    `;
    const buttonHoverCSS = `background: #30363d;`;

    const panelHTML = `
      <div style="${headerCSS}">
        <span>📱 Catálogo Cuballama</span>
        <div>
          <button id="cuballama-copy-btn" style="${buttonCSS}">Copiar</button>
          <button id="cuballama-close-btn" style="${buttonCSS}">Cerrar</button>
        </div>
      </div>
      <pre style="${contentCSS}">${textContent}</pre>
    `;
    
    const panel = document.createElement("div");
    panel.setAttribute('data-darkreader-ignore', '');
    panel.id = PANEL_ID;
    panel.style.cssText = panelCSS;
    panel.innerHTML = panelHTML;
    document.body.appendChild(panel);
    
    const copyBtn = panel.querySelector("#cuballama-copy-btn");
    const closeBtn = panel.querySelector("#cuballama-close-btn");
    
    copyBtn.onmouseenter = () => copyBtn.style.cssText = buttonCSS + buttonHoverCSS;
    copyBtn.onmouseleave = () => copyBtn.style.cssText = buttonCSS;
    
    copyBtn.addEventListener("click", () => {
      navigator.clipboard.writeText(textContent).then(() => {
        copyBtn.textContent = "Copiado ✓";
        setTimeout(() => (copyBtn.textContent = "Copiar"), 2000);
      }).catch(err => console.error("Error al copiar:", err));
    });
    
    closeBtn.addEventListener("click", () => panel.remove());
  }

  // -------------------------------------------------
  // 🚀 FUNCIÓN PRINCIPAL (IIFE)
  // -------------------------------------------------
  async function run() {
    try {
      const rawToken = localStorage.getItem(TOKEN_KEY);
      const authToken = normalizeAuth(rawToken);
      
      if (!authToken) {
        console.warn(`⚠️ No se encontró el token en localStorage con la llave: '${TOKEN_KEY}'`);
      }
      
      const response = await fetch(API_URL, {
        headers: authToken ? { "Authorization": authToken } : {}
      });

      if (!response.ok) {
        throw new Error(`Error HTTP: ${response.status} ${response.statusText}`);
      }
      
      const data = await response.json();
      console.log("📦 Respuesta de la API recibida:", data);

      const rawOffers = getOffersFromData(data);
      if (!rawOffers || rawOffers.length === 0) {
        console.warn("⚠️ No se encontraron ofertas en la respuesta de la API.");
        return;
      }

      const normalizedOffers = rawOffers.map(normalizeOffer);
      const catalogText = formatOffersToText(normalizedOffers);
      
      console.log("--- CATÁLOGO GENERADO ---");
      console.log(catalogText);
      
      displayCatalogPanel(catalogText);

    } catch (error) {
      console.error("❌ Ocurrió un error en la ejecución del script:", error);
    }
  }

  run();

})();
```

Con esto tu catálogo se arma automáticamente y cada línea detecta palabras clave para añadir el emoji correcto.
